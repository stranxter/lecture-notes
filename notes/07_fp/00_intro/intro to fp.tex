\begin{frame}[fragile]
  \frametitle{Характеристика на езика}
  
\begin{itemize}
  \item Чисто функционален език
  \item Строго типизиран език
  \begin{lstlisting}[basicstyle=\small, language=Haskell]
    5 + "5"
    f :: [Int] -> Int
    f [] = 0
    f _ = 1
    \end{lstlisting}
  \item Лениво (мързеливо) оценяване
  \begin{lstlisting}[basicstyle=\small, language=Haskell]
    f x = x : f (x+1)
    take 5 (f 0)
  \end{lstlisting}
\end{itemize}
